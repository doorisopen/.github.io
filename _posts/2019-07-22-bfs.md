#include <iostream>
#include <queue>
#include <vector>

using namespace std;

int	num = 7;
// 방문 처리 하기 위한 배열 
int c[7];
vector<int> a[8];
void bfs(int start){
	queue<int> q;
	q.push(start);
	c[start] = true;
	while(!q.empty()){
		int x = q.front();
		q.pop();
		printf("%d ", x);
		for(int i = 0; i < a[x].size(); i++){
			int y = a[x][i];
			// 방문한 상태가 아니라면 큐에 넣어준다.
			if(!c[y]){
				q.push(y);
				c[y] = true; 
			}
		}
		
	}
	
}
int main(void){
	a[1].push_back(2);
	a[2].push_back(1);

	a[5].push_back(4);

	bfs(1);
	
	return 0;
}
