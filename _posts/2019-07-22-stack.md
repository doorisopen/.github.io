#### C++ STL Library를 사용한 Stack 구현


{% highlight javascript %}
#include <iostream>
#include <stack>
using namespace std;

int main(void){
	stack<int> s;
	
	s.push(8);
	s.push(5);
	s.push(2);
	s.pop();
	s.push(7);
	s.pop();
	while(!s.empty()){
		cout << s.top() << ' ';
		s.pop();
	}
	
	return 0;
}
